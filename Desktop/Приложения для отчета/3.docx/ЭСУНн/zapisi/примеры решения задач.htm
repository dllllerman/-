6.    Примеры решения задач

Массивы записей

Условие задачи 1. Заполнить массив данными о книгах магазина (инвентарный номер, автор, название, цена, количество) и вывести полученный массив на экран в виде таблицы с заголовками столбцов.

Листинг 1

program magazin;

uses crt;

const f=5;

type book=record

num:word;

avtor: string[20];

title:string[30];

price:real;

kol:integer;

end;

var    m:array [1..f] of book;

i:integer;

begin

clrscr;

{Заполнение массивами данными о книгах} 

for i:=1 to f do

begin

writeln('Введите сведения о книгах');

writeln('Книга ',i,':');

write('Инвентарный номер->');read(m[i].num);

write('Автор->');read(m[i].avtor);

write('Название->');read(m[i].title);

write('Цена->');read(m[i].price);

write('Количество->');read(m[i].kol);

end;

{Вывод на экран содержимого таблицы}

clrscr;

textcolor(12);

writeln('Сведения о наличии книг в магазине');

textcolor(2);

writeln('Инв.номер':10,'Автор':15,'Название':15,'Цена':8,'Кол-во':8);

textcolor(15);

for i:=1 to f do

begin

{writeln(m[i].num:10,m[i].avtor:15,

m[i].title:15,m[i].price:8:2,m[i].kol:8);}

with m[i] do

writeln(num:10,avtor:15,title:15,price:8:2,kol:8);

end;

end.

Файлы записей

Условие задачи 1. Создать на диске файл с именем magazin.dat и записать в него сведения о книгах магазина (инвентарный номер, автор, название, цена, количество экземпляров). Конец записи – инвентарный номер равный 0. Вывести на экран содержимое этого файла в виде таблицы.

Заполнение файла данными

Листинг 2

program magazine_1;

uses crt;

type  book=record

num:word;

avtor: string[20];

title:string[20];

price:real;

kol:integer;

end;

var    buf:book;

name:string[12];

f:file of book;

begin

clrscr;

write('Имя файла:');

read(name);

assign(f,name+’.dat’);

rewrite(f);

repeat

writeln('Введите сведения о книгах');

with buf do

begin

write('Инв.номер:'); read(num);

if num<>0 then

begin

write('Автор->');read(avtor);

write('Название->');read(title);

write('Цена->');read(price);

write('Количество->');read(kol);

write(f,buf);

end;

end;

clrscr;

until buf.num=0;

writeln('Запись в файл завершена');

close(f);

end.

Вывод на экран содержимого файла в виде таблицы

program magazine_2;

uses crt;

type  book=record

num:word;

avtor, title: string[20];

price:real;

kol:integer;

end;

var    buf:book;

name:string[12];

f:file of book;

begin

clrscr;

write('Имя файла->');

read(name);

assign(f,name+’.dat’);

reset(f);

textcolor(12);

writeln('Сведения о наличии книг в магазине');

textcolor(2);

writeln('Инв.номер':10,'Автор':15,'Название':15,'Цена':8,'Кол-во':8);

textcolor(15);

while not eof(f) do

begin

with buf do

begin

read(f,buf);

writeln(num:10,avtor:15,title:15,price:8:2,kol:8);

end;

end;

writeln('Все!!!');

close(f);

end.
